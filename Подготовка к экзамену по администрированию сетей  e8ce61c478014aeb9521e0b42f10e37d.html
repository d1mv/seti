      1. systemctl enable --now bind
      1. vim /etc/bind/options.conf
        1. listen-on { any; };
          forwarders { 94.232.137.104; };
          dnssec-validation no;
          recursion yes;
          allow-query { any; };
          allow-recursion { any; };
      1. vim /etc/bind/local.conf
        1. добавляем после слов Add other zones here:
          zone "au.team" {
          type master;
          file "au.team";
          allow-transfer {20.20.20.100;};
          };
          zone "10.10.10.in-addr.arpa" {
          type master;
          file "left.reverse";
          allow-transfer {20.20.20.100;};
          };
          zone "20.20.20.in-addr.arpa" {
          type master;
          file "right.reverse";
          allow-transfer {20.20.20.100;};
          };
          zone "35.35.35.in-addr.arpa" {
          type master;
          file "cli.reverse";
          allow-transfer {20.20.20.100;};
          };
      1. cd /etc/bind/zone/
      1. cp localhost au.team
      1. vim au.team
        1. заменяем localhost. на au.team. и root.localhost. на root.au.team.
        1. пишем через табуляцию зоны:
      1. cp localhost right.reverse
      1. vim right.reverse
        1. заменяем localhost. на 20.20.20.in-addr.arpa. и root.localhost. на root.20.20.20.in-addr.arpa.
        1. пишем через табуляцию зоны:
      1. cp right.reverse left.reverse
      1. vim left.reverse
        1. заменяем 20.20.20.in-addr.arpa. на 10.10.10.in-addr.arpa. и root.localhost. на root.10.10.10.in-addr.arpa.
        1. пишем через табуляцию зоны:
      1. cp right.reverse cli.reverse
      1. vim cli.reverse
        1. заменяем 10.10.10.in-addr.arpa. на 35.35.35.in-addr.arpa. и root.localhost. на root.35.35.35.in-addr.arpa.
        1. пишем через табуляцию зоны:
      1. chmod 777 au.team
      1. chmod 777 right.reverse
      1. chmod 777 left.reverse
      1. chmod 777 cli.reverse
      1. systemctl restart bind
      1. vim /etc/resolv.conf
        1. должен быть указан только один nameserver 127.0.0.1
    • WEB-R
      1. systemctl enable --now bind
      1. vim /etc/bind/options.conf
        1. что должно быть в options:
          listen-on { any; };
          forwarders { 10.10.10.100; };
          dnssec-validation no;
          recursion yes;
          allow-query { any; };
          allow-recursion { any; };
      1. vim /etc/bind/local.conf
        1. добавляем после слов Add other zones here:
          zone "au.team" {
          type slave;
          file "slave/au.team";
          masters {10.10.10.100;};
          };
          zone "10.10.10.in-addr.arpa" {
          type slave;
          file "slave/left.reverse";
          masters {10.10.10.100;};
          };
          zone "20.20.20. in-addr.arpa" {
          type slave;
          file "slave/right.reverse";
          masters {10.10.10.100;};
          };
          zone "35.35.35. in-addr.arpa" {
          type slave;
          file "slave/cli.reverse";
          masters {10.10.10.100;};
          };
      1. chown named:named /var/lib/bind/zone/slave/
      1. chown named:named /etc/bind/zone/slave/
      1. systemctl restart bind
      1. vim /etc/resolv.conf
        1. должен быть указан только один nameserver 127.0.0.1

  • ПУНКТ 7. ФАЙЛОВОЕ ХРАНИЛИЩЕ
    • SRV-L
      1. lsblk
      1. cfdisk /dev/sdb → gpt → NEW → TYPE → Linux RAID → WRITE → yes → QUIT
      1. С пункта 2 повторить действия со всеми остальными дисками (sdb, sdc, sdd, sde)
      1. mdadm --create /dev/md0 --level=5 --raid-devices=4 /dev/sdb1 /dev/sdc1 /dev/sdd1 /dev/sde1
      1. mdadm --detail --scan --verbose | tee -a /etc/mdadm.conf
      1. mkfs.ext4 /dev/md0
      1. mkdir /raid5
      1. vim /etc/fstab
        1. добавить в конец, пишем через табуляцию:
          /dev/md0 /raid5 ext4 defaults 0 0
      1. reboot
      1. systemctl restart nfs
      1. mkdir /raid5/nfs
      1. chmod 777 /raid5/nfs
      1. vim /etc/exports
        1. в конец добавляем:
          /raid5/nfs 10.10.10.110(rw,sync) 20.20.20.100(rw,sync)

        1. 10.10.10.110-адрес WEB-L
          20.20.20.100-адрес WEB-R
    • WEB-L
      1. mkdir /mnt/nfs
      1. vim /etc/fstab
        1. добавляем в конец, пишем через табуляцию
          10.10.10.100:/raid5/nfs /mnt/nfs nfs rw,sync 0 0
    • WEB-R
      1. mkdir /mnt/nfs
      1. vim /etc/fstab
        1. добавляем в конец, пишем через табуляцию
          10.10.10.100:/raid5/nfs /mnt/nfs nfs rw,sync 0 0
